// For songs:
var items = []



boxes = [...document.querySelectorAll('.page_charts_section_charts_item_info_main_left')]
for (var i in boxes) {
  let box = boxes[i]
  let song = [...box.querySelectorAll('.song')].map(el => el.textContent.trim())[0]
  const artist = [...box.querySelectorAll('.artist .ui_name_locale')][0].innerText
  const genre = [...box.querySelectorAll('a.genre.comma_separated.first')][0].innerText
  const item = {
    "song": song,
    "artist": artist,
    "genre": genre
  }
  items.push(item)
}
